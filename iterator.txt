class Libro {
    var titulo

    func obtenerTitulo(){
        return obtenerTitulo(titulo)
    }
}

class Biblioteca {

    var libros[]

    func agregarLibros(libro: Libro){
        return  libros.append(libro)
    }

    func cantidadLibros() {
        return libros.count()
    }

    func obtenerLibros(posicion) {
        return libros[posicion]
    }
}

class BibliotecaIterador {

    var indice = 0
    var biblioteca = Biblioteca()

    func hayMasLibros() {
        return indice < biblioteca.cantidadLibros()
    }

    func libroSiguiente() {
        if hayMasLibros() {
            var libro = biblioteca.obtenerLibros(indice)
            indice += 1
            return libro
        } else {
            return nill
        }
    }
}
