/*
Categoria: Creacional
Proposito: Crear nuevos objetos copiando un prototipo existente en lugar de
crear objetos desde cero.
Como funciona:
El objeto a copiar tiene un metodo clonar que devuelve una copia exacta.
Reduce el costo de crear objetos complejos.
Ejemplo:
Clonar documentos con contenido similar
*/


//Interfaz Prototype
interfaz Prototipo:
    metodo clonar()


//Clases concretas
clase Guerrero implementa Prototipo:

    constructor(nombre, nivel, arma):
        this.nombre = nombre
        this.nivel = nivel
        this.arma = arma
    
    metodo clonar():
        return new Guerrero(this.nombre, this.nivel, this.arma)
    
    metodo mostrar():
        imprimir "Guerrero: " + this.nombre + ", Nivel: " + this.nivel + ", Arma: " + this.arma
    

//Uso del patron
guerreroOriginal = new Guerrero("Arthas", 10, "Espada")

guerreroClonado = guerreroOriginal.clonar()
guerreroClonado.nombre = "Lancelot"

guerreroOriginal.mostrar()   // Guerrero: Arthas, Nivel: 10, Arma: Espada
guerreroClonado.mostrar()    // Guerrero: Lancelot, Nivel: 10, Arma: Espada
