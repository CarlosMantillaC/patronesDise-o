/*
Categoria: Estructural
Proposito: Separar la abstraccion de su implementacion para que ambas
puedan variar independientemente.
Como funciona:
Hay dos jerarquias: una para la abstraccion, otra para la implementacion.
La abstraccion mantiene una referencia a la implementacion.
El puente es la referencia a la implementacion desde la abstraccion.
Ejemplo: Un control remoto (abstraccion) que puede funcionar con diferentes
dispositivos (implementacion).
*/

//Implementador
interfaz Dispositivo:
    metodo encender()
    metodo apagar()

//Implementaciones concretas
clase Televisor implementa Dispositivo:
    metodo encender():
        imprimir "Televisor encendido"

    metodo apagar():
        imprimir "Televisor apagado"

clase Radio implementa Dispositivo:
    metodo encender():
        imprimir "Radio encendido"

    metodo apagar():
        imprimir "Radio apagado"


//Abstraccion
clase ControlRemoto:
    constructor(dispositivo: Dispositivo):
        this.dispositivo = dispositivo

    metodo encender():
        dispositivo.encender()
    
    metodo apagar():
        dispositivo.apagar()


//Uso del patron
tv = new Televisor()
radio = new Radio()

controlTV = ControlRemoto(tv)
controlRadio = ControlRemoto(radio)

controlTV.encender()
controlTV.apagar()

controlRadio.encender()
controlRadio.apagar()
