/*
Categoria: Creacional
Proposito: Proveer un conjunto de metodos abstractos para crear familias de 
objetos relacionados sin especificar sus clases concretas.
Como funciona: 
Se crea una fabrica abstracta con metodos para crear distintos productos.
Cada fabrica concreta implementa esos metodos para crear productos relacionados.
Ejemplo: 
Un sistema que crea interfaces graficas para distintas plataformas, donde una
fabrica crea botones, menus y ventanas para windows, y otra para Mac.
*/

//Interfaces de productos
interfaz Boton:
    metodo dibujar()

interfaz Checkbox:
    metodo dibujar()


//Productos concretos para tema claro
clase BotonClaro implementa Boton:
    metodo dibujar():
        return "Dibujando boton claro"

clase CheckboxClaro implementa Checkbox:
    metodo dibujar():
        return "Dibujando checkbox claro"


//Productos concretos para tema oscuro
clase BotonOscuro implementa Boton:
    metodo dibujar():
        return "Dibujando boton oscuro"

clase CheckboxOscuro implementa Checkbox:
    metodo dibujar():
        return "Dibujando checkbox oscuro"


//Fabrica Abstracta
interfaz FabricaGUI: 
    metodo crearBoton()
    metodo crearCheckbox()


//Fabricas Concretas:
clase FabricaTemaClaro implementa FabricaGUI:
    metodo crearBoton(): 
        return new BotonClaro()
    metodo crearCheckbox():
        return new CheckboxClaro()

clase FabricaTemaOscuro implementa FabricaGUI:
    metodo crearBoton():
        return new BotonOscuro()
    metodo crearCheckbox():
        return new CheckboxOscuro()


//Cliente que usa la fabrica
clase Aplicacion:
    constructor(temaUsuario)
        fabrica = FabricaGUI

        si temaUsuario == "claro":
            fabrica = new FabricaTemaClaro
        sino:
            fabrica = new FabricaTemaOscuro

        this.boton = fabrica.crearBoton()
        this.checkbox = fabrica.crearCheckbox()
    
    metodo dibujarInterfaz():
        boton.dibujar()
        boton.checkbox()


//Uso del patron
temaUsuario = "oscuro"
app = new Aplicacion(temaUsuario)
app.dibujarInterfaz()
